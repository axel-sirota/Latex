\section{Problemas}

\begin{question}
	Problema1
\end{question}

\begin{solution}
	Vayamos por partes:
	
	\begin{enumerate}
		\item Sea $f \in \rm {Bil}(V,W)$ y llamemos $\psi: V \times W \rightarrow F(V,W)$ al morfismo $\psi(v,w) = (v,w)$ que aplica sobre el espcio libre. Luego por la propiedad universal de los objetos libres existe una \'unica $f^{\ast} : F(V,W) \rightarrow U$ tal que $f^{\ast} \circ \psi = f$. Notemos que $R(V,W) \subset \ker f^{\ast}$ por lo tanto por la propiedad universal del cociente existe una \'unica $\tilde{f} : V \otimes W := \quotient{F(V,W)}{R(V,W)} \rightarrow U$ tal que $\tilde{f} \circ \pi = f^{\ast}$, con lo que $\tilde{f} \circ \pi \circ \psi  = \tilde{f} \circ (\pi \circ \psi) = \tilde{f} \circ \phi =f^{\ast} \circ \psi = f$ y es trivialmente lineal.
		\item En efecto, notemos el siguiente diagrama:
		
		\[
		\begin{tikzcd}
			A \arrow[rd] \arrow[r, "\phi"] & B \\
			& C
		\end{tikzcd}
		\]
		
	\end{enumerate}
	
\end{solution}
